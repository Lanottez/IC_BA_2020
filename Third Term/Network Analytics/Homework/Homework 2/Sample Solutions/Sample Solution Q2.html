Calculate the distance matrix using approximation¶

In [3]:
"""
For a pair of places, calculate the distance between them based on spherical coordinates.

Args:
    points = pandas dataframe containing all points, 
    i = index of the first city, 
    j = index of the second city

Returns:
    The distance between two points on the Earth
"""

def distance(points, i, j):
    lat1 = points['latitude'][i] 
    lat2 = points['latitude'][j]
    long1 = points['longitude'][i]
    long2 = points['longitude'][j]
    
    # Convert latitude and longitude to
    # spherical coordinates in radians.
    degrees_to_radians = pi / 180.0
    
    # phi = 90 - latitude
    phi1 = (90.0 - lat1)*degrees_to_radians
    phi2 = (90.0 - lat2)*degrees_to_radians
    
    # theta = longitude
    theta1 = long1 * degrees_to_radians
    theta2 = long2 * degrees_to_radians
 
    # Compute spherical distance from spherical coordinates.
    #
    # For two locations in spherical coordinates
    # (1, theta, phi) and (1, theta', phi')
    # cosine( arc length ) =
    # sin phi sin phi' cos(theta-theta') + cos phi cos phi'
    # distance = rho * arc length
    
    cos1 = (sin(phi1)*sin(phi2)*cos(theta1 - theta2) + cos(phi1)*cos(phi2))
    cos1 = round(cos1,8)
    
    # Remember to multiply arc by the radius of the earth
    # in your favorite set of units to get the length.
    arc = acos( cos1 ) * 6373 
 
    return arc

c. Challenge problem, finding an optimal tour using Gurobi (Solution from Group 2)¶

We connect with the Python interface of Gurobi and find an optimal tour using the sub-tour elimination integer programming formulation.

In [8]:
"""
    Add the sub-tour elimination constraints. This function calls 'subtour' to 
    verify the optimal tour we found in previous solutions, if previous solution 
    has subtour, add another subtour elimination constraint and repeat solving 
    process until a an optimal tour without subtours is found 

"""

def subtourelim(model, where):
    
    if where == GRB.callback.MIPSOL:
        selected = []
        # make a list of edges selected in the solution
        for i in range(n):
            sol = model.cbGetSolution([model._vars[i,j] for j in range(n)])
            selected += [(i,j) for j in range(n) if sol[j] > 0.5]

        # find the shortest cycle in the selected edge list
        tour = subtour(selected)
        if len(tour) < n:
            # add a subtour elimination constraint
            expr = 0
            for i in range(len(tour)):
                for j in range(i+1, len(tour)):
                    expr += model._vars[tour[i], tour[j]]
            model.cbLazy(expr <= len(tour)-1)

'''
    Given a list of edges, find the shortest tour, if the length of shortest tour is not equal to the number of nodes, there
    exists subtour and need add another constraint
'''
            
def subtour(edges): 

    visited = [False]*n
    cycles = []
    lengths = []
    selected = [[] for i in range(n)]
  
    for x,y in edges:
        selected[x].append(y)
    while True:
        current = visited.index(False)
        thiscycle = [current]
        while True:
            visited[current] = True
            neighbors = [x for x in selected[current] if not visited[x]]
            if len(neighbors) == 0:
                break
            current = neighbors[0]
            thiscycle.append(current)
        cycles.append(thiscycle)
        lengths.append(len(thiscycle))
        if sum(lengths) == n:
            break
            
    return cycles[lengths.index(min(lengths))]
