summary <- crime_map %>% 
  group_by(region) %>% 
  summarise(mean_long = mean(long),
            mean_lat = mean(lat),
            Assault = first(Assault)) %>% 
  filter(!is.na(Assault))

# US map
us_map <- ggplot()+
  geom_polygon(data = crime_map, aes(x=long, y=lat, group = group), fill = "gray90", colour = "gray30", size = 0.25)+
