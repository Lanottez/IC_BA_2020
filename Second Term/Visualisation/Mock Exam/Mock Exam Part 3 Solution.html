A example using the gridExtra package

library(gridExtra)
## 
## Attaching package: 'gridExtra'
## The following object is masked from 'package:dplyr':
## 
##     combine
p1 <- ggplot(iris) +
  geom_density(aes(x = Sepal.Length, color = Species))

p2 <- ggplot(iris) +
  geom_density(aes(x = Petal.Length, color = Species))

grid.arrange(p1, p2, ncol = 2)

A example using just the tidyverse package

# make the subset of data into tidy data
df <- iris %>% 
  select(Species, Sepal.Length, Petal.Length) %>% 
  gather(key = "measure", value = "value", Sepal.Length, Petal.Length ) %>% 
  mutate(measure = as.factor(measure))

ggplot(df) +
  geom_density(aes(x = value, color = Species))+
  facet_grid(. ~ measure)
